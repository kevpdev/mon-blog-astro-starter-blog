## Layout Templates

### Blog Post Layout
```astro
---
// BlogLayout.astro
interface Props {
  title: string;
  description: string;
  pubDate: Date;
  heroImage?: string;
  tags?: string[];
}

const { title, description, pubDate, heroImage, tags = [] } = Astro.props;
---

<BaseLayout {title} {description}>
  {heroImage && <img src={heroImage} alt={title} />}
  <h1>{title}</h1>
  <time datetime={pubDate.toISOString()}>{pubDate.toLocaleDateString()}</time>
  <div class="tags">
    {tags.map(tag => <span class="tag">{tag}</span>)}
  </div>
  <main>
    <slot />
  </main>
</BaseLayout>
```
